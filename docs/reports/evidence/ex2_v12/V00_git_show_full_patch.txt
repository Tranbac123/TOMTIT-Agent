    feat(EX2): add capability-aware skill catalog
    
    Introduce DisabledSkillReason, DisabledSkill, and SkillCatalog. The
    catalog partitions all three static built-in SkillSpecs against the
    exact resolved ToolRegistry: compatible skills enter the strict active
    SkillRegistry; incompatible skills produce DisabledSkill records with
    sorted missing-tool evidence.
    
    SkillAwareIntentPlanner now receives SkillCatalog (not SkillRegistry)
    and routes intents to mapped, unavailable, or unknown outcomes.
    Disabled intents return a deterministic capability-unavailable FINISH
    plan instead of falling through to the unknown path.
    
    builtin_skill_specs() becomes argument-free; filtering moves entirely
    into SkillCatalog.from_specs(). SkillRegistry.from_specs() remains
    strict (MissingSkillToolError if any required tool is absent).
    
    Remote backend result: 1 active skill (WEB_SEARCH), 2 disabled with
    exact missing-tool evidence (write_note, read_note). Local: 3 active,
    0 disabled.

+@dataclass(frozen=True)
+class SkillCatalog:
+    """Partitions all built-in SkillSpecs into active (compatible) and disabled
+    (missing tools) sets against the exact resolved ToolRegistry.
+
+    Construction follows the spec §5.7 order:
+    1. materialize specs once
+    2. validate duplicate SkillName
+    3. validate duplicate intent ownership
+    4. compute missing tools vs exact resolved ToolRegistry
+    5. partition into compatible/incompatible
+    6. build strict active SkillRegistry

+    def __post_init__(self) -> None:

+        idx: dict[IntentName, DisabledSkill] = {}
+        for ds in self.disabled:
+            for intent in ds.applicable_intents:
+                idx[intent] = ds
+        object.__setattr__(self, "_disabled_intent_index", MappingProxyType(idx))

+    @classmethod
+    def from_specs(
+        cls,

+        *,
+        tools: Mapping[ToolName, ToolSpec],
+    ) -> SkillCatalog:
+        # Step 1: materialize
+        specs_list = list(specs)
+
+        # Step 2+3: validate duplicates across ALL specs before partitioning
+        seen_names: set[SkillName] = set()
+        seen_intents: dict[IntentName, SkillName] = {}
+        for spec in specs_list:
+            if spec.name in seen_names:
+                raise DuplicateSkillError(
+                    f"Skill {spec.name!r} defined more than once"
+                )
+            seen_names.add(spec.name)
